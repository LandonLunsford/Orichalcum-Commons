	function getTweener(target:Object, propertyName:String, propertyValue:*):ITweener
	{
		const tweener:ITweener = _tweenerPlugins[propertyName];
		return tweener
			? tweener
			: getDefaultTweener(_defaultTweeners[getQualifiedClassName(propertyValue)]);
	}
	
	/**
	 * this implementation only allows one tweener per property
	 */
	function install(tweener:ITweener, propertyOrProperties:*):void
	{
		if (tweener == null)
			throw new ArgumentError();
			
		if (propertyOrProperties is String)
		{
			_tweenerPlugins[propertyOrProperties] = tweener;
		}
		else if (propertyOrProperties is Array || propertyOrProperties is Vector.<String>)
		{
			for each(var propertyName:String in propertyOrProperties)
				_tweenerPlugins[propertyName] = tweener;
		}
	}
